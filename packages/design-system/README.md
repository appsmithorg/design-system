## Install

```bash
npm install @appsmith/design-system
```
or
```bash
yarn add @appsmithorg/design-system
```

## Usage

```jsx
import { TooltipComponent } from "@appsmithorg/design-system";
ReactDOM.render(<TooltipComponent />, mountNode);
```
